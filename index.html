async loadPGNFile(event) { const url = event.target.value; if (!url) { return; } try { const response = await fetch(url); if (!response.ok) { throw new Error(`Failed to fetch PGN file from ${url}`); } const pgnContent = await response.text(); this.parsePGN(pgnContent); } catch (error) { alert(`Error loading PGN file: ${error.message}`); console.error('Error loading PGN file:', error); } }
